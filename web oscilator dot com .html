    <script>

      // =============================

      // p5.js Sketch (Simplified Example)

      // =============================

      // Replace this with your full sketch code if desired.

      

      let gridSize = 400;        // Overall grid size

      let numCells = 20;         // Number of cells per side

      let cellSize = gridSize / numCells;

      let time = 0.0;

      let rotationY = 0.0;

      let waveAmplitude = 50.0;

      let waveFrequency = 0.5;

      let rotationSpeedY = 0.005;

      let scaleAmount = 0.2;

      let waveSpeed = 1.5;

      let points = [];


      function setup() {

        createCanvas(windowWidth, windowHeight, WEBGL);

        colorMode(HSB, 360, 100, 100);

        // Initialize grid points

        for (let i = 0; i <= numCells; i++) {

          points[i] = [];

          for (let j = 0; j <= numCells; j++) {

            points[i][j] = createVector(0, 0, 0);

          }

        }

      }


      function draw() {

        background(0, 0, 10); // Dark background using HSB (low brightness)

        time += 0.016;

        rotationY += rotationSpeedY;

        rotateY(rotationY);

        let centerOffset = gridSize / 2.0;

        // Update grid vertices using a wave effect

        for (let i = 0; i <= numCells; i++) {

          for (let j = 0; j <= numCells; j++) {

            let baseX = i * cellSize - centerOffset;

            let baseY = j * cellSize - centerOffset;

            let z = waveAmplitude * 

                    (sin(baseX * waveFrequency * 0.1 + time * waveSpeed) +

                     cos(baseY * waveFrequency * 0.1 - time * waveSpeed * 0.7)) / 2.0;

            points[i][j].set(baseX, baseY, z);

          }

        }

        // Dynamic stroke color that cycles over time

        let hueVal = (millis() / 10) % 360;

        stroke(hueVal, 80, 100);

        strokeWeight(1.5);

        noFill();

        // Draw horizontal grid lines

        for (let j = 0; j <= numCells; j++) {

          beginShape();

          for (let i = 0; i <= numCells; i++) {

            let pt = points[i][j];

            vertex(pt.x, pt.y, pt.z);

          }

          endShape();

        }

        // Draw vertical grid lines

        for (let i = 0; i <= numCells; i++) {

          beginShape();

          for (let j = 0; j <= numCells; j++) {

            let pt = points[i][j];

            vertex(pt.x, pt.y, pt.z);

          }

          endShape();

        }

      }


      // Handle resizing the canvas when the window size changes

      function windowResized() {

        resizeCanvas(windowWidth, windowHeight);

      }
